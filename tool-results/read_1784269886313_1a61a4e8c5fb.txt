   125→function SignatureCanvas({ onSave, width = 400, height = 150 }: { onSave: (dataUrl: string) => void; width?: number; height?: number }) {
   126→  const canvasRef = useRef<HTMLCanvasElement>(null);
   127→  const isDrawing = useRef(false);
   128→  const lastPos = useRef({ x: 0, y: 0 });
   129→
   130→  const getPos = useCallback((e: React.MouseEvent | React.TouchEvent) => {
   131→    const canvas = canvasRef.current!;
   132→    const rect = canvas.getBoundingClientRect();
   133→    const scaleX = canvas.width / rect.width;
   134→    const scaleY = canvas.height / rect.height;
   135→    if ('touches' in e) {
   136→      return { x: (e.touches[0].clientX - rect.left) * scaleX, y: (e.touches[0].clientY - rect.top) * scaleY };
   137→    }
   138→    return { x: (e.clientX - rect.left) * scaleX, y: (e.clientY - rect.top) * scaleY };
   139→  }, []);
   140→
   141→  const startDraw = useCallback((e: React.MouseEvent | React.TouchEvent) => {
   142→    isDrawing.current = true;
   143→    const pos = getPos(e);
   144→    lastPos.current = pos;
   145→    const ctx = canvasRef.current!.getContext('2d')!;
   146→    ctx.beginPath();
   147→    ctx.moveTo(pos.x, pos.y);
   148→  }, [getPos]);
   149→
   150→  const draw = useCallback((e: React.MouseEvent | React.TouchEvent) => {
   151→    if (!isDrawing.current) return;
   152→    const pos = getPos(e);
   153→    const ctx = canvasRef.current!.getContext('2d')!;
   154→    ctx.strokeStyle = '#1a1a2e';
   155→    ctx.lineWidth = 2.5;
   156→    ctx.lineCap = 'round';
   157→    ctx.lineJoin = 'round';
   158→    ctx.beginPath();
   159→    ctx.moveTo(lastPos.current.x, lastPos.current.y);
   160→    ctx.lineTo(pos.x, pos.y);
   161→    ctx.stroke();
   162→    lastPos.current = pos;
   163→  }, [getPos]);
   164→
   165→  const endDraw = useCallback(() => { isDrawing.current = false; }, []);
   166→
   167→  const clear = () => {
   168→    const ctx = canvasRef.current!.getContext('2d')!;
   169→    ctx.clearRect(0, 0, canvasRef.current!.width, canvasRef.current!.height);
   170→  };
   171→
   172→  const save = () => {
   173→    const dataUrl = canvasRef.current!.toDataURL('image/png');
   174→    onSave(dataUrl);
   175→  };
   176→
   177→  return (
   178→    <div className="space-y-3">
   179→      <canvas
   180→        ref={canvasRef}
   181→        width={width}
   182→        height={height}
   183→        className="border-2 border-dashed border-muted-foreground/30 rounded-lg cursor-crosshair bg-white w-full"
   184→        style={{ touchAction: 'none' }}
   185→        onMouseDown={startDraw}
   186→        onMouseMove={draw}
   187→        onMouseUp={endDraw}
   188→        onMouseLeave={endDraw}
   189→        onTouchStart={startDraw}
   190→        onTouchMove={draw}
   191→        onTouchEnd={endDraw}
   192→      />
   193→      <div className="flex gap-2">
   194→        <Button variant="outline" size="sm" onClick={clear} className="flex-1">
   195→          <X className="w-4 h-4 mr-1" /> Clear
   196→        </Button>
   197→        <Button size="sm" onClick={save} className="flex-1 bg-emerald-600 hover:bg-emerald-700">
   198→          <Check className="w-4 h-4 mr-1" /> Apply
   199→        </Button>
   200→      </div>
   201→    </div>
   202→  );
   203→}

   205→function TypeSignature({ onSave }: { onSave: (dataUrl: string) => void }) {
   206→  const [text, setText] = useState('');
   207→  const canvasRef = useRef<HTMLCanvasElement>(null);
   208→
   209→  useEffect(() => {
   210→    if (!canvasRef.current) return;
   211→    const ctx = canvasRef.current.getContext('2d')!;
   212→    ctx.clearRect(0, 0, canvasRef.current.width, canvasRef.current.height);
   213→    if (!text) return;
   214→    ctx.font = 'italic 42px "Segoe Script", "Apple Chancery", cursive, serif';
   215→    ctx.fillStyle = '#1a1a2e';
   216→    ctx.textBaseline = 'middle';
   217→    ctx.fillText(text, 10, canvasRef.current.height / 2);
   218→  }, [text]);
   219→
   220→  const save = () => {
   221→    if (!text.trim()) return;
   222→    onSave(canvasRef.current!.toDataURL('image/png'));
   223→  };
   224→
   225→  return (
   226→    <div className="space-y-3">
   227→      <Input
   228→        placeholder="Type your name"
   229→        value={text}
   230→        onChange={(e) => setText(e.target.value)}
   231→      />
   232→      <canvas ref={canvasRef} width={400} height={80} className="border rounded-lg bg-white w-full" />
   233→      <Button size="sm" onClick={save} disabled={!text.trim()} className="w-full bg-emerald-600 hover:bg-emerald-700">
   234→        <Check className="w-4 h-4 mr-1" /> Apply
   235→      </Button>
   236→    </div>
   237→  );
   238→}

   240→function UploadSignature({ onSave }: { onSave: (dataUrl: string) => void }) {
   241→  const [preview, setPreview] = useState<string | null>(null);
   242→
   243→  const handleFile = (e: React.ChangeEvent<HTMLInputElement>) => {
   244→    const file = e.target.files?.[0];
   245→    if (!file) return;
   246→    const reader = new FileReader();
   247→    reader.onload = () => {
   248→      const result = reader.result as string;
   249→      setPreview(result);
   250→    };
   251→    reader.readAsDataURL(file);
   252→  };
   253→
   254→  const save = () => {
   255→    if (preview) onSave(preview);
   256→  };
   257→
   258→  return (
   259→    <div className="space-y-3">
   260→      <div className="border-2 border-dashed border-muted-foreground/30 rounded-lg p-6 text-center">
   261→        {preview ? (
   262→          <img src={preview} alt="Signature preview" className="max-h-24 mx-auto" />
   263→        ) : (
   264→          <div>
   265→            <ImageIcon className="w-8 h-8 mx-auto mb-2 text-muted-foreground" />
   266→            <p className="text-sm text-muted-foreground">Upload PNG or JPG</p>
   267→          </div>
   268→        )}
   269→        <input type="file" accept="image/png,image/jpeg" onChange={handleFile} className="mt-2 text-sm" />
   270→      </div>
   271→      <Button size="sm" onClick={save} disabled={!preview} className="w-full bg-emerald-600 hover:bg-emerald-700">
   272→        <Check className="w-4 h-4 mr-1" /> Apply
   273→      </Button>
   274→    </div>
   275→  );
   276→}
